CREATE TABLE public.audit_log (
  id integer GENERATED ALWAYS AS IDENTITY NOT NULL,
  actor uuid,
  action text NOT NULL,
  details jsonb,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT audit_log_pkey PRIMARY KEY (id),
  CONSTRAINT audit_log_actor_fkey FOREIGN KEY (actor) REFERENCES public.users(id)
);
CREATE TABLE public.game_lines (
  id integer GENERATED ALWAYS AS IDENTITY NOT NULL,
  game_id uuid NOT NULL,
  source text NOT NULL DEFAULT 'fanduel'::text,
  spread_team_id integer NOT NULL,
  spread_value numeric NOT NULL,
  fetched_at timestamp with time zone NOT NULL DEFAULT now(),
  is_active_line boolean NOT NULL DEFAULT false,
  CONSTRAINT game_lines_pkey PRIMARY KEY (id),
  CONSTRAINT game_lines_game_id_fkey FOREIGN KEY (game_id) REFERENCES public.games(id),
  CONSTRAINT game_lines_spread_team_id_fkey FOREIGN KEY (spread_team_id) REFERENCES public.teams(id)
);
CREATE TABLE public.games (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  week_id integer NOT NULL,
  external_game_id text UNIQUE,
  commence_time timestamp with time zone NOT NULL,
  home_team_id integer NOT NULL,
  away_team_id integer NOT NULL,
  status text NOT NULL DEFAULT 'scheduled'::text,
  final_scores jsonb,
  CONSTRAINT games_pkey PRIMARY KEY (id),
  CONSTRAINT games_week_id_fkey FOREIGN KEY (week_id) REFERENCES public.weeks(id),
  CONSTRAINT games_home_team_id_fkey FOREIGN KEY (home_team_id) REFERENCES public.teams(id),
  CONSTRAINT games_away_team_id_fkey FOREIGN KEY (away_team_id) REFERENCES public.teams(id)
);
CREATE TABLE public.pick_settlement (
  user_id uuid NOT NULL,
  game_id uuid NOT NULL,
  pick_id uuid,
  points_delta integer CHECK (points_delta >= '-20'::integer AND points_delta <= 10),
  outcome USER-DEFINED,
  graded_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT pick_settlement_pkey PRIMARY KEY (user_id, game_id),
  CONSTRAINT pick_settlement_game_id_fkey FOREIGN KEY (game_id) REFERENCES public.games(id),
  CONSTRAINT pick_settlement_pick_id_fkey FOREIGN KEY (pick_id) REFERENCES public.picks(id)
);
CREATE TABLE public.picks (
  user_id uuid NOT NULL,
  game_id uuid NOT NULL,
  picked_team_id integer NOT NULL,
  weight USER-DEFINED NOT NULL,
  locked_at timestamp with time zone NOT NULL,
  locked_line_id integer,
  locked_spread_team_id integer NOT NULL,
  locked_spread_value numeric NOT NULL,
  locked_by uuid NOT NULL DEFAULT auth.uid(),
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  CONSTRAINT picks_pkey PRIMARY KEY (id),
  CONSTRAINT picks_user_id_fkey FOREIGN KEY (user_id) REFERENCES public.users(id),
  CONSTRAINT picks_game_id_fkey FOREIGN KEY (game_id) REFERENCES public.games(id),
  CONSTRAINT picks_picked_team_id_fkey FOREIGN KEY (picked_team_id) REFERENCES public.teams(id)
);
CREATE TABLE public.results (
  game_id uuid NOT NULL,
  winning_team_id integer,
  cover_result text,
  graded_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT results_pkey PRIMARY KEY (game_id),
  CONSTRAINT results_game_id_fkey FOREIGN KEY (game_id) REFERENCES public.games(id)
);
CREATE TABLE public.seasons (
  id integer GENERATED ALWAYS AS IDENTITY NOT NULL,
  league text NOT NULL DEFAULT 'NFL'::text,
  year integer NOT NULL,
  CONSTRAINT seasons_pkey PRIMARY KEY (id)
);
CREATE TABLE public.settings (
  id boolean NOT NULL DEFAULT true,
  odds_api_monthly_cap integer NOT NULL DEFAULT 500,
  odds_api_calls_used_current_month integer NOT NULL DEFAULT 0,
  reset_on date,
  missed_pick_penalty integer DEFAULT '-1'::integer,
  CONSTRAINT settings_pkey PRIMARY KEY (id)
);
CREATE TABLE public.teams (
  id integer GENERATED ALWAYS AS IDENTITY NOT NULL,
  league text NOT NULL DEFAULT 'NFL'::text,
  external_key text UNIQUE,
  name text NOT NULL UNIQUE,
  short_name text NOT NULL,
  CONSTRAINT teams_pkey PRIMARY KEY (id)
);
CREATE TABLE public.totals (
  user_id uuid NOT NULL,
  week_id integer NOT NULL,
  points_delta integer NOT NULL DEFAULT 0,
  season_total_cached integer NOT NULL DEFAULT 0,
  CONSTRAINT totals_pkey PRIMARY KEY (user_id, week_id),
  CONSTRAINT totals_user_id_fkey FOREIGN KEY (user_id) REFERENCES public.users(id),
  CONSTRAINT totals_week_id_fkey FOREIGN KEY (week_id) REFERENCES public.weeks(id)
);
CREATE TABLE public.users (
  id uuid NOT NULL,
  display_name text NOT NULL,
  role text NOT NULL DEFAULT 'player'::text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT users_pkey PRIMARY KEY (id),
  CONSTRAINT users_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id)
);
CREATE TABLE public.weeks (
  id integer GENERATED ALWAYS AS IDENTITY NOT NULL,
  season_id integer NOT NULL,
  week_number integer NOT NULL,
  start_ts timestamp with time zone NOT NULL,
  end_ts timestamp with time zone NOT NULL,
  CONSTRAINT weeks_pkey PRIMARY KEY (id),
  CONSTRAINT weeks_season_id_fkey FOREIGN KEY (season_id) REFERENCES public.seasons(id)
);
